# ======================================
# Conda Environment Setup and Commands
# ======================================

# Create new environment with Python 3.10
conda create -p venv python==3.10

# Activate the environment (absolute path)
conda activate D:\Projects\study-complete-data-science-machine-learning-dl-nlp-bootcamp\venv

# OR (relative path)
conda activate venv/

# Install required packages from requirements.txt
pip install -r requirements.txt
# (Make sure the file name is spelled correctly)

# Install Jupyter kernel
pip install ipykernel

# ======================================
# Maintenance & Troubleshooting Commands
# ======================================

# Update Conda to the latest version
conda update -n base -c defaults conda

# Install or upgrade pip manually
python -m ensurepip --upgrade
python -m pip install --upgrade pip

# Use Python's -m flag if pip.exe fails
python -m pip install -r requirements.txt

# Check versions
python --version
pip --version

# Deactivate the Conda environment
conda deactivate
